 var firstTimeReturn = 0;
 function trackProductCategories(entry,parent){
 	var category=entry;
 	var str=category.replace("[","");
 	str=str.replace("]","");	
 	var splits=str.split(", ");
 	$.each(splits, function (index, value) {
 		if(index!=0){
 		if(firstTime == 0){
 			parent["subsection"+index] = [];
 		}
 		}
 		});
 	$.each(splits, function (index, value) {
 		if(index!=0){
 		parent["subsection"+index].push($.trim(value));
 		firstTimeReturn = 1;
 		}
 		});
 }
